340 int Mario::collision_direction(pro2::Rect r){
341  pro2::Rect last_rm, rm;
342  rm = get_rect();
343  last_rm = {last_pos_.x-width/2,last_pos_.y-height+1,last_pos_.x+width/2,last_pos_.y };
344 
345  bool is_within_x = (rm.left >= r.left and rm.left <= r.right) or (rm.right >= r.left and rm.right <= r.right);
346  bool is_within_y = (rm.top >= r.top and rm.top <= r.bottom) or (rm.bottom >= r.top and rm.bottom <= r.bottom);
347  bool up_crossed = last_rm.top <= r.bottom and rm.top >= r.bottom;
348  bool down_crossed = last_rm.bottom <= r.top and rm.bottom >= r.top;
349  bool left_crossed = last_rm.right <= r.left and rm.right >= r.left;
350  bool right_crossed = last_rm.left >= r.right and rm.left <= r.right;
351  if(is_within_x and up_crossed) return 1; //UP
352  if(is_within_y and left_crossed) return 2; //LEFT
353  if(is_within_y and right_crossed) return 3; //RIGHT
354  if(is_within_x and down_crossed) return 4; // DOWN
355  else return 0;
